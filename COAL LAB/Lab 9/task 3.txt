INCLUDE Irvine32.inc
.data
	MSG BYTE "Enter value : ",0
	MSG1 BYTE "SQAURE  :",0
.code
main PROC
	mov edx,OFFSET MSG
	call writestring
	CALL LOCALSUARE
exit
main ENDP

LOCALSUARE PROC
	LOCAL VAR:DWORD
	ENTER 0,0
	MOV eax,0
	CALL READDEC
	CALL Crlf
	MOV VAR,eax
	mul VAR
	mov edx,OFFSET MSG1
	call writestring
	CALL WRITEDEC
	CALL Crlf
	LEAVE
	ret
LOCALSUARE ENDP
END main
