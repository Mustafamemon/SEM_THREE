//
#include "stdafx.h"
#include <iostream>
using namespace std;
extern "C"
{
	void asmPROC(int[]);
}
int main()
{
	int arr[11]={0};
	asmPROC(arr);
	cout<<"Fibonacci series : ";
	for(int i=0;i<10;i++)
	{
		cout<<arr[i]<<" ";
	}
	cout<<endl;
}
