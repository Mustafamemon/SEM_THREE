Title Lab_6
INCLUDE Irvine32.inc

.data
	var DWORD 0
	msg1 BYTE "HELLO",0
	msg2 BYTE "WORLD",0
.code
main PROC
	mov ecx,10
	mov eax,var
	mov ebx,2
	l1:
		mov edx,0
		div ebx
		CMP edx,0
		JE hello
			mov edx,OFFSET msg2
			call writeString
			call crlf
			jmp f
		hello:
			mov edx,OFFSET msg1
			call writeString
			call crlf
		f:
			INC var
			mov eax,var
	CMP var,ecx
	JLE l1
	
exit
main ENDP
END main
