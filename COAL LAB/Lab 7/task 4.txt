Title Lab_6
INCLUDE Irvine32.inc

.data
	arr	WORD	10, 4, 7, 14, 299, 156, 3, 19, 29, 300, 20
	msg1 BYTE "EXIST!",0
	msg2 BYTE "NOT EXIST!",0
.code
main PROC
	mov ecx,LENGTHOF arr
	call readint 
	mov ebx,0
l1:
	CMP arr[esi],ax
	JE common
	add esi,TYPE arr
loop l1
	mov edx,OFFSET msg2
	call writeString
	call crlf
exit
common:
	mov edx,OFFSET msg1
	call writeString
	call crlf
exit
main ENDP
END main
