INCLUDE Irvine32.inc

.data
	Array BYTE 1,2,3,4,5,6,7,8

.code
main PROC
	push DWORD PTR 3
	push OFFSET Array
	call LOADS
	mov ecx,LENGTHOF Array
	mov edi,OFFSET Array
	mov eax,0
l1:
	mov al,[edi]
	call writedec
	call crlf
	INC edi
loop l1
exit
main endp
LOADS PROC,
	A:DWORD , B:DWORD
	mov edi,A
	mov esi,edi
	INVOKE Str_length,A
	mov ecx,eax
	mov eax,0
	cld
l1:
	lodsb
	mul B
	stosb
loop l1
ret
LOADS endp
end main
