.data
	string1 BYTE "MUSTAFA",0
	string2 BYTE "MANGA",0
	msg BYTE "EQUAL",0
	msg1 BYTE "NOT-",0
.code
main PROC
	push OFFSET string1
	push OFFSET string2
	call MOVE
	pop edi
	pop edx
	call writestring
	call crlf
	mov edx,edi
	call writestring
exit
main endp
MOVE PROC
	push ebp
	mov ebp,esp
	mov edi,[ebp+8]
	mov esi,[ebp+12]
	INVOKE Str_length,[ebp+8]
	mov ecx,eax
	cld
rep	movsb
	mov esp,ebp
	pop ebp
ret
MOVE endp
end main
